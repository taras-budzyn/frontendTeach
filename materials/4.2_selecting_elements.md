  ```
    <p>
      First paragraph
    </p>
    <p>

    </p>
    <div>
      <p>
        First inner paragraph
      </p>
      <p>
        Second inner paragraph
      </p>
    </div>
  </div>
  <div id="second">
    <p>
      First paragraph in second div
    </p>
    <p>
      Second paragraph in second div
    </p>
    <p>
      Third paragraph in second div
    </p>
  </div>
  <div class="last-div">
    <p>
      First paragraph in last div
    </p>
    <p>
      Second paragraph in last div
    </p>
  </div>
  ```
